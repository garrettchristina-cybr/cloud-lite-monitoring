Demo runbook (CLI)

# Start or rebuild
docker compose up -d --build

# Full demo (brute + 5xx + AI summary)
powershell -ExecutionPolicy Bypass -File .\scripts\demo-all.ps1

# Brute force only (customize IP/Count/Delay)
powershell -ExecutionPolicy Bypass -File .\scripts\demo-bruteforce.ps1 -Ip "203.0.113.50" -Count 12 -DelayMs 500

# HTTP 5xx demo only
powershell -ExecutionPolicy Bypass -File .\scripts\demo-5xx.ps1

# AI summary to Discord + summary.txt
python ai/ai_summarize.py
